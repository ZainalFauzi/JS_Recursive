Buatlah rekursive function untuk menghitung angka faktorial Contoh perhitungan angka faktorial: 4! = 4 * 3! 4! = 4 * 3 * 2! 4! = 4 * 3 * 2 * 1 4! = 24 
            function factorial (q){
                if (q == 0) {
                    return 1
                }
                else { 
                    factorial(q-1) 
                    return q * factorial(q-1)
                }
            }
            console.log(factorial(4))
